defineRegisters: 
    mov r1,#0x008844       	//Snake colour (green)
	mov r2,#0xffffff       //Background colour (white)
    mov r3, #527           //Tail position, initialised
    mov r4, #528         	 //Head position, initialised
drawSnake:
    	str r1,[r3]             	//Tail
    	str r1,[r4]             	//Head
moveSnake:
    str r2,[r3]          	//Reset tail to Background
    add r3,r3,#1      	//Increment the tail pointer by 1
    add r4,r4,#1        	//Increment the head pointer by 1
    str r1,[r4]         	//Draw new head
    b moveSnake		//Loop
